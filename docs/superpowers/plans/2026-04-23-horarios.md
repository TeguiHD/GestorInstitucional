```typescript
import { IsInt, IsOptional, IsString, IsUUID, Matches, Max, Min } from 'class-validator';

export class UpsertSlotDto {
  @ApiProperty()
  @IsUUID()
  courseId!: string;

  @ApiProperty()
  @IsUUID()
  subjectId!: string;

  @IsInt()
```

```typescript
import { Injectable } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service.js';
import type { UpsertSlotDto } from './dto/upsert-slot.dto.js';

@Injectable()
export class ScheduleService {
  constructor(private readonly prisma: PrismaService) {}

  async findByCourse(courseId: string, year: number, semester: number) {
    return this.prisma.scheduleSlot.findMany({
      where: { courseId, year, semester },
      include: {
        subject: { select: { id: true, name: true, code: true } },
        teacher: { select: { id: true, firstName: true, lastName: true } },
      },
      orderBy: [{ dayOfWeek: 'asc' }, { slotNumber: 'asc' }],
    });
  }

  async findByTeacher(teacherId: string, year: number, semester: number) {
    return this.prisma.scheduleSlot.findMany({
      where: { teacherId, year, semester },
      include: {
        subject: { select: { id: true, name: true, code: true } },
        course: { select: { id: true, name: true, code: true } },
      },
      orderBy: [{ dayOfWeek: 'asc' }, { slotNumber: 'asc' }],
    });
  }

  async upsert(dto: UpsertSlotDto) {
    return this.prisma.scheduleSlot.upsert({
      where: {
        courseId_dayOfWeek_slotNumber_year_semester: {
          courseId: dto.courseId,
          dayOfWeek: dto.dayOfWeek,
          slotNumber: dto.slotNumber,
          year: dto.year,
          semester: dto.semester,
        },
      },
      create: {
        courseId: dto.courseId,
        subjectId: dto.subjectId,
        teacherId: dto.teacherId,
        dayOfWeek: dto.dayOfWeek,
        slotNumber: dto.slotNumber,
        startTime: dto.startTime,
        endTime: dto.endTime,
        year: dto.year,
        semester: dto.semester,
        ...(dto.roomNumber ? { roomNumber: dto.roomNumber } : {}),
      },
      update: {
        subjectId: dto.subjectId,
        teacherId: dto.teacherId,
        startTime: dto.startTime,
        endTime: dto.endTime,
        ...(dto.roomNumber !== undefined ? { roomNumber: dto.roomNumber } : {}),
      },
      include: {
        subject: { select: { id: true, name: true, code: true } },
        teacher: { select: { id: true, firstName: true, lastName: true } },
      },
    });
  }

  async removeSlot(
    courseId: string,
    dayOfWeek: number,
    slotNumber: number,
    year: number,
    semester: number,
  ) {
    return this.prisma.scheduleSlot.deleteMany({
      where: { courseId, dayOfWeek, slotNumber, year, semester },
    });
  }
}
```

```typescript
import {
  Body,
  Controller,
  Delete,
  Get,
  HttpCode,
  HttpStatus,
  Param,
  Put,
  Query,
  UseGuards,
} from '@nestjs/common';
import { ApiBearerAuth, ApiOperation, ApiTags } from '@nestjs/swagger';
import { SystemRole } from '@prisma/client';

import { JwtAuthGuard } from '../common/guards/jwt-auth.guard.js';
import { RolesGuard } from '../common/guards/roles.guard.js';
import { Roles } from '../common/decorators/roles.decorator.js';
import { ScheduleService } from './schedule.service.js';
import { UpsertSlotDto } from './dto/upsert-slot.dto.js';

@ApiTags('schedule')
@UseGuards(JwtAuthGuard, RolesGuard)
@ApiBearerAuth()
@Controller('schedule')
export class ScheduleController {
  constructor(private readonly schedule: ScheduleService) {}

  @Get('course/:courseId')
  @ApiOperation({ summary: 'Horario semanal de un curso' })
  findByCourse(
    @Param('courseId') courseId: string,
    @Query('year') year: number,
    @Query('semester') semester: number,
  ) {
    return this.schedule.findByCourse(courseId, Number(year), Number(semester));
  }

  @Get('teacher/:teacherId')
  @ApiOperation({ summary: 'Horario semanal de un docente' })
  findByTeacher(
    @Param('teacherId') teacherId: string,
    @Query('year') year: number,
    @Query('semester') semester: number,
  ) {
    return this.schedule.findByTeacher(teacherId, Number(year), Number(semester));
  }

  @Put('slot')
  @Roles(SystemRole.SUPER_ADMIN, SystemRole.DIRECTOR, SystemRole.UTP)
  @ApiOperation({ summary: 'Asignar o actualizar un bloque horario' })
  upsert(@Body() dto: UpsertSlotDto) {
    return this.schedule.upsert(dto);
  }

  @Delete('course/:courseId/slot')
  @HttpCode(HttpStatus.NO_CONTENT)
  @Roles(SystemRole.SUPER_ADMIN, SystemRole.DIRECTOR, SystemRole.UTP)
  @ApiOperation({ summary: 'Eliminar un bloque horario' })
  removeSlot(
    @Param('courseId') courseId: string,
    @Query('day') day: number,
    @Query('slot') slot: number,
    @Query('year') year: number,
    @Query('semester') semester: number,
  ) {
    return this.schedule.removeSlot(
      courseId,
      Number(day),
      Number(slot),
      Number(year),
      Number(semester),
    );
  }
}
```

```tsx
import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query';
import { X } from 'lucide-react';
import { useState } from 'react';
import { toast } from 'sonner';
import { api } from '@/lib/api';
import { cn } from '@/lib/cn';
import { useUser } from '@/stores/auth.store';
import { useEffectiveSchoolId } from '@/stores/school.store';

const DAYS = ['Lunes', 'Martes', 'Miércoles', 'Jueves', 'Viernes'];
const SLOTS = [1, 2, 3, 4, 5, 6, 7, 8];

const DEFAULT_TIMES: Record<number, { start: string; end: string }> = {
  1: { start: '08:00', end: '08:45' },
  2: { start: '08:45', end: '09:30' },
  3: { start: '09:45', end: '10:30' },
  4: { start: '10:30', end: '11:15' },
  5: { start: '11:30', end: '12:15' },
  6: { start: '12:15', end: '13:00' },
  7: { start: '14:00', end: '14:45' },
  8: { start: '14:45', end: '15:30' },
};

type Slot = {
  id: string;
  dayOfWeek: number;
  slotNumber: number;
  startTime: string;
  endTime: string;
  roomNumber?: string;
  subject: { id: string; name: string; code: string };
  teacher: { id: string; firstName: string; lastName: string };
};

type Subject = { id: string; name: string; code: string };
type Teacher = { id: string; firstName: string; lastName: string; email: string };

const SUBJECT_COLORS = [
  'bg-blue-100 text-blue-800 dark:bg-blue-900/30 dark:text-blue-300',
  'bg-green-100 text-green-800 dark:bg-green-900/30 dark:text-green-300',
  'bg-purple-100 text-purple-800 dark:bg-purple-900/30 dark:text-purple-300',
  'bg-orange-100 text-orange-800 dark:bg-orange-900/30 dark:text-orange-300',
  'bg-pink-100 text-pink-800 dark:bg-pink-900/30 dark:text-pink-300',
  'bg-teal-100 text-teal-800 dark:bg-teal-900/30 dark:text-teal-300',
  'bg-yellow-100 text-yellow-800 dark:bg-yellow-900/30 dark:text-yellow-300',
  'bg-red-100 text-red-800 dark:bg-red-900/30 dark:text-red-300',
];

export function ScheduleTab({ courseId }: { courseId: string }) {
  const qc = useQueryClient();
  const user = useUser();
  const schoolId = useEffectiveSchoolId();
  const isAdmin = user?.roles.some((r) => ['SUPER_ADMIN', 'DIRECTOR', 'UTP'].includes(r)) ?? false;

  const year = new Date().getFullYear();
  const [semester, setSemester] = useState(new Date().getMonth() < 7 ? 1 : 2);
  const [editCell, setEditCell] = useState<{ day: number; slot: number } | null>(null);
  const [form, setForm] = useState({ subjectId: '', teacherId: '', roomNumber: '' });

  const { data: slots } = useQuery<Slot[]>({
    queryKey: ['schedule', courseId, year, semester],
    queryFn: () => api.get(`/schedule/course/${courseId}?year=${year}&semester=${semester}`),
  });

  const { data: subjects } = useQuery<Subject[]>({
    queryKey: ['subjects', courseId],
    queryFn: () => api.get(`/subjects?courseId=${courseId}`),
    // fetch for all roles — used by legend even in read-only mode
  });

  const { data: teachers } = useQuery<Teacher[]>({
    queryKey: ['teachers', schoolId],
    queryFn: () => api.get(`/users?schoolId=${schoolId}&roles=PROFESOR`),
    enabled: isAdmin,
  });

  const subjectColorMap = (() => {
    const map: Record<string, string> = {};
    let i = 0;
    for (const slot of slots ?? []) {
      if (!map[slot.subject.id]) {
        map[slot.subject.id] = SUBJECT_COLORS[i % SUBJECT_COLORS.length]!;
        i++;
      }
    }
    return map;
  })();

  const getSlot = (day: number, slot: number) =>
    slots?.find((s) => s.dayOfWeek === day && s.slotNumber === slot);

  const upsertMut = useMutation({
    mutationFn: () => {
      if (!editCell || !form.subjectId || !form.teacherId)
        throw new Error('Completa todos los campos');
      return api.put('/schedule/slot', {
        courseId,
        subjectId: form.subjectId,
        teacherId: form.teacherId,
        dayOfWeek: editCell.day,
        slotNumber: editCell.slot,
        startTime: DEFAULT_TIMES[editCell.slot]!.start,
        endTime: DEFAULT_TIMES[editCell.slot]!.end,
        year,
        semester,
        ...(form.roomNumber ? { roomNumber: form.roomNumber } : {}),
      });
    },
    onSuccess: () => {
      toast.success('Bloque asignado');
      setEditCell(null);
      setForm({ subjectId: '', teacherId: '', roomNumber: '' });
      void qc.invalidateQueries({ queryKey: ['schedule', courseId, year, semester] });
    },
    onError: (e: unknown) => toast.error((e as Error).message),
  });

  const deleteMut = useMutation({
    mutationFn: ({ day, slot }: { day: number; slot: number }) =>
      api.del(
        `/schedule/course/${courseId}/slot?day=${day}&slot=${slot}&year=${year}&semester=${semester}`,
      ),
    onSuccess: () => {
      toast.success('Bloque eliminado');
      void qc.invalidateQueries({ queryKey: ['schedule', courseId, year, semester] });
    },
  });

  return (
    <div className="space-y-4">
      {/* Semester selector */}
      <div className="flex items-center gap-2">
        <span className="text-sm text-muted-foreground">Semestre:</span>
        {[1, 2].map((s) => (
          <button
            key={s}
            onClick={() => setSemester(s)}
            className={cn(
              'px-3 py-1.5 text-xs font-medium rounded-lg border transition-colors',
              semester === s
                ? 'bg-primary text-primary-foreground border-primary'
                : 'border-border text-muted-foreground hover:bg-muted',
            )}
          >
            {s}° Semestre
          </button>
        ))}
        {isAdmin && (
          <span className="text-xs text-muted-foreground ml-2">
            Haz clic en un bloque para editar
          </span>
        )}
      </div>

      {/* Schedule grid */}
      <div className="rounded-xl border border-border bg-background overflow-hidden">
        <div className="overflow-x-auto">
          <table className="w-full text-xs">
            <thead>
              <tr className="bg-muted/50">
                <th className="text-left px-3 py-2.5 w-24 text-muted-foreground font-semibold uppercase tracking-wide">
                  Bloque
                </th>
                {DAYS.map((d) => (
                  <th
                    key={d}
                    className="text-center px-2 py-2.5 font-semibold text-muted-foreground uppercase tracking-wide min-w-[120px]"
                  >
                    {d}
                  </th>
                ))}
              </tr>
            </thead>
            <tbody>
              {SLOTS.map((slotNum) => (
                <tr key={slotNum} className="border-t border-border">
                  <td className="px-3 py-2 text-muted-foreground">
                    <div className="font-medium">B{slotNum}</div>
                    <div className="text-[10px]">
                      {DEFAULT_TIMES[slotNum]!.start}–{DEFAULT_TIMES[slotNum]!.end}
                    </div>
                  </td>
                  {[1, 2, 3, 4, 5].map((day) => {
                    const slot = getSlot(day, slotNum);
                    const isEditing = editCell?.day === day && editCell?.slot === slotNum;
                    return (
                      <td key={day} className="px-1 py-1">
                        {slot ? (
                          <div
                            className={cn(
                              'rounded-lg px-2 py-1.5 relative group cursor-pointer transition-all',
                              subjectColorMap[slot.subject.id] ?? SUBJECT_COLORS[0],
                              isAdmin && 'hover:opacity-80',
                            )}
                            onClick={() => {
                              if (!isAdmin) return;
                              setEditCell({ day, slot: slotNum });
                              setForm({
                                subjectId: slot.subject.id,
                                teacherId: slot.teacher.id,
                                roomNumber: slot.roomNumber ?? '',
                              });
                            }}
                          >
                            <div className="font-semibold text-[11px] leading-tight">
                              {slot.subject.code}
                            </div>
                            <div className="text-[10px] opacity-80 leading-tight">
                              {slot.teacher.lastName}
                            </div>
                            {slot.roomNumber && (
                              <div className="text-[10px] opacity-60">Sala {slot.roomNumber}</div>
                            )}
                            {isAdmin && (
                              <button
                                onClick={(e) => {
                                  e.stopPropagation();
                                  deleteMut.mutate({ day, slot: slotNum });
                                }}
                                className="absolute top-0.5 right-0.5 opacity-0 group-hover:opacity-100 p-0.5 rounded hover:bg-black/20"
                              >
                                <X className="size-2.5" />
                              </button>
                            )}
                          </div>
                        ) : isAdmin ? (
                          <button
                            onClick={() => {
                              setEditCell({ day, slot: slotNum });
                              setForm({ subjectId: '', teacherId: '', roomNumber: '' });
                            }}
                            className="w-full h-10 rounded-lg border border-dashed border-border hover:border-primary hover:bg-primary/5 transition-colors flex items-center justify-center text-muted-foreground hover:text-primary"
                          >

                className="flex-1 text-sm px-4 py-2 rounded-lg bg-primary text-primary-foreground disabled:opacity-50"
              >
                {upsertMut.isPending ? 'Guardando…' : 'Guardar'}
              </button>
              <button
                onClick={() => setEditCell(null)}
                className="flex-1 text-sm px-4 py-2 rounded-lg border border-border hover:bg-muted"
              >
                Cancelar
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Legend */}
      {Object.keys(subjectColorMap).length > 0 && subjects && (
        <div className="flex flex-wrap gap-2">
          {subjects
            .filter((s) => subjectColorMap[s.id])
            .map((s) => (
              <span
                key={s.id}
                className={cn('px-2 py-1 rounded-lg text-xs font-medium', subjectColorMap[s.id])}
              >
                {s.code} — {s.name}
              </span>
            ))}
        </div>
      )}
    </div>
  );
}
```
